+from flask import Flask, request
+import random
+import statistics
+
+# 全域變數濫用
+app = Flask(__name__)
+DATA = []
+RESULTS = {}
+
+# 魔術數字
+LIMIT = 37
+
+@app.route("/")
+def home():
+    # 無意義註解
+    # 這裡是首頁
+    return "Welcome to Bad Flask App!"
+
+@app.route("/generate")
+def generate():
+    global DATA
+    # 沒有任何錯誤處理
+    DATA = [random.randint(1, 100) for _ in range(LIMIT)]
+    return f"Generated {len(DATA)} numbers"
+
+@app.route("/analyze")
+def analyze():
+    global DATA, RESULTS
+    if len(DATA) == 0:
+        return "No data yet"
+    # 過度巢狀 + 重複邏輯
+    if len(DATA) > 5:
+        meanVal = statistics.mean(DATA)
+        RESULTS["mean"] = meanVal
+        RESULTS["meanAgain"] = statistics.mean(DATA)  # 重複計算
+        if meanVal > 50:
+            RESULTS["flag"] = "HIGH"
+        else:
+            RESULTS["flag"] = "LOW"
+    if len(DATA) > 10:
+        RESULTS["median"] = statistics.median(DATA)
+        RESULTS["medianPlus42"] = statistics.median(DATA) + 42  # 魔術數字
+    return str(RESULTS)
+
+@app.route("/clear")
+def clear():
+    global DATA, RESULTS
+    DATA = []
+    RESULTS = {}
+    return "Cleared everything!"
+
+if __name__ == "__main__":
+    # 無意義註解
+    # 這裡是主程式入口
+    app.run(debug=True, port=5000)

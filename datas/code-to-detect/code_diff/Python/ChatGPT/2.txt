+import pygame
+import random
+import time
+import math
+
+# 初始化散落在檔案各處
+pygame.init()
+
+SCREEN_W, SCREEN_H = 640, 480
+screen = pygame.display.set_mode((SCREEN_W, SCREEN_H))
+pygame.display.set_caption("Totally Fine Game")
+
+clock = pygame.time.Clock()
+
+# 全域狀態：改來改去才刺激
+STATE = {
+    "running": True,
+    "score": 0,
+    "player": [SCREEN_W // 2, SCREEN_H // 2],
+    "velocity": 3,
+    "color": [255, 255, 255],
+    "last_time": time.time()
+}
+
+
+def do_everything(event=None):
+    # 函式名稱完全不透露它在幹嘛
+    if event and event.type == pygame.KEYDOWN:
+        STATE["velocity"] += random.choice([-1, 0, 1])
+
+    # 神秘計算
+    now = time.time()
+    delta = now - STATE["last_time"]
+    STATE["last_time"] = now
+
+    if delta > 0:
+        STATE["score"] += int(delta * 10) % 7
+
+    # 顏色在漂，但理由不明
+    for i in range(3):
+        STATE["color"][i] = (STATE["color"][i] + random.randint(-5, 5)) % 256
+
+
+def move_player(keys):
+    # 重複邏輯但不一樣寫法
+    if keys[pygame.K_LEFT]:
+        STATE["player"][0] -= STATE["velocity"]
+    if keys[pygame.K_RIGHT]:
+        STATE["player"][0] += int(math.sqrt(STATE["velocity"] ** 2))
+    if keys[pygame.K_UP]:
+        STATE["player"][1] -= abs(STATE["velocity"])
+    if keys[pygame.K_DOWN]:
+        STATE["player"][1] += STATE["velocity"] or 1
+
+    # 邊界檢查寫一半
+    STATE["player"][0] %= SCREEN_W
+    STATE["player"][1] %= SCREEN_H
+
+
+def draw_stuff():
+    # 每幀重新填色，但顏色不是常數
+    screen.fill((0, 0, STATE["score"] % 255))
+
+    # 玩家其實是個圓，但資料結構是 list
+    pygame.draw.circle(
+        screen,
+        STATE["color"],
+        STATE["player"],
+        10 + STATE["score"] % 15
+    )
+
+    # 字型每次都 new，一點都不浪費
+    font = pygame.font.SysFont(None, 24)
+    text = font.render(f"Score-ish: {STATE['score']}", True, (255, 255, 255))
+    screen.blit(text, (10, 10))
+
+
+while STATE["running"]:
+    # event loop + game logic 全混在一起
+    for event in pygame.event.get():
+        if event.type == pygame.QUIT:
+            STATE["running"] = False
+        do_everything(event)
+
+    keys = pygame.key.get_pressed()
+    move_player(keys)
+
+    draw_stuff()
+
+    pygame.display.flip()
+
+    # magic number frame limit
+    clock.tick(57)
+
+
+# 退出時才想到要 quit
+pygame.quit()
